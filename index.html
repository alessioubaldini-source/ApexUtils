Esempi Pro — Pattern Reali Oracle APEX
Pattern 1 · Calcolo Live (Debounce) + UI Logic
// [Input P1_VALORE] - Key Release -> Execute JS Code
var runCalc = ApexUtils.debounce(function() {
  ApexUtils.runProcess('CALCOLA_KPI', {
    val: $v('P1_VALORE')
  }).then(function(res) {
    // Output su item
    ApexUtils.safeSetValue('P1_RISULTATO', res.score);

    // Logica UI condizionale
    if (res.valid) {
      ApexUtils.enableItems('#btnConferma');
      ApexUtils.debug('Valore valido: ' + res.score);
    } else {
      ApexUtils.disableItems('#btnConferma');
      ApexUtils.debug(res.error_msg);
    }
  });
}, 300);
Pattern 2 · Update Items & Sync Grid
// [Button "Carica Dettagli"]
ApexUtils.updateItemsFromProcess('GET_DATA_DETT', ['P1_ID'])
  .then(function() {
    // Items popolati (P1_NOME, P1_STATUS...)

    // Sincronizza riga master per riflettere modifiche
    ApexUtils.syncIgSelectedRecords('MASTER_IG');

    ApexUtils.debug('Dati aggiornati');
  });
Pattern 3 · Workflow Sequenziale + Refresh UI
// [Button "Esegui Chiusura"]
ApexUtils.runSequentialProcesses([
  { name: 'STEP_1_VALIDA' },
  { name: 'STEP_2_CALCOLA' },
  { name: 'STEP_3_ARCHIVIA' }
], function(results) {

  // Refresh parallelo dashboard
  ApexUtils.refreshRegions(['GRID_LOG', 'CHART_STATS'])
    .then(function() {
      // Abilita e switch su Tab Report
      ApexUtils.setTabState('TAB_REPORT', true);
      
      ApexUtils.debug('Chiusura completata!');
    });
});
Pattern 4 · Mappatura colonna IG → Page Item
// Mappatura: Colonna IG -> Item di Pagina
const mapping = {
  "TC033_T_MOD_LIQ_CPI":          "P4037_C_MOD_LIQ_CPI_INF",
  "TC033_T_TRC":                  "P4037_T_TRC_INF",
  "TC033_C_COD_ACC_SUP":          "P4037_C_COD_ACC_SUP_INF",
  "TC033_C_COD_CND_RIG_ACC_SUP":  "P4037_C_COD_CND_RIG_ACC_SUP_INF",
  "TC033_T_NOT":                  "P4037_T_NOT_INF",
  "TC033_T_OPE_SGL_CPI":          "P4037_INF_T_OPE_SGL_INF",
  "TC033_C_OPE_SGL_CPI":          "P4037_INF_C_OPE_SGL_INF",
  "TC033_T_OPE_SGL_CPI_SUP":      "P4037_INF_T_OPE_SGL_CPI_SUP",
  "TC033_C_OPE_SGL_CPI_SUP":      "P4037_INF_C_OPE_SGL_CPI_SUP",
  "TC033_C_TIP_DOC":              "P4037_C_TIP_DOC_INF",
  "TC033_C_PER_LIQ":              "P4037_C_PER_LIQ_CMS",
  "TC033_C_PER_LIQ_CPI":          "P4037_C_PER_LIQ_CPI",
  "TC033_N_VAL_SGL_INF":          "P4037_INF_N_VAL_SGL_INF",
  "TC033_N_VAL_SGL_SUP":          "P4037_INF_N_VAL_SGL_SUP",
  "TC033_T_UNI_MIS_CPI":          "P4037_INF_C_UNI_MIS_SGL",
  "C_COD_CND_RIG_ACC":            "P4037_C_COD_CND_RIG_ACC"
};

// Itera sulla mappa per recuperare i valori dalla griglia e impostarli negli item
for (const [columnName, itemName] of Object.entries(mapping)) {
  // Recupera il valore.
  // getIgEventRecordValue estrae correttamente il valore '.v' se è una LOV (pOnlyCode=true di default)
  let val = ApexUtils.getIgEventRecordValue(this, columnName);
  
  // Imposta il valore in modo sicuro (gestisce null e sopprime eventi change)
  ApexUtils.safeSetValue(itemName, val);
}

// Esegue il refresh della griglia TC054 con un ritardo di 300ms
setTimeout(() => {
  ApexUtils.refreshIg("TC054");
}, 300);
